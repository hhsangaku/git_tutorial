'''git

Gitチュートリアル

2024/12/30

# ワークツリーのファイルをステージに追加する
git add
git add

# ステージのファイルをリポジトリに追加する
git commit

# コミット前に差分を表示する
git commit -v

# ワークツリーとステージのadd状況, ステージとリポジトリのコミット状況を比較する
git status

# ワークツリーとステージのファイル差分を表示
git diff
git diff

# ステージとリポジトリのファイル差分を表示
git diff --staged

# ログを確認する
git log

# 1行分で内容表示
git log --oneline

# ファイルの変更差分を表示する
git log -p

#表示するコミット回数を指定する
git log -n

'''
