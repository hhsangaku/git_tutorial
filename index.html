'''git

Gitチュートリアル

2024/12/30

# ワークツリーのファイルをステージに追加する
git add
git add

# ステージのファイルをリポジトリに追加する
git commit

# コミット前に差分を表示する
git commit -v

# ワークツリーとステージのadd状況, ステージとリポジトリのコミット状況を比較する
git status

# ワークツリーとステージのファイル差分を表示
git diff
git diff

# ステージとリポジトリのファイル差分を表示
git diff --staged

# ログを確認する
git log

# 1行分で内容表示
git log --oneline

# ファイルの変更差分を表示する
git log -p

#表示するコミット回数を指定する
git log -n

# 削除（ローカル＜ワークツリー＞とリポジトリ双方の削除を行う）
git rm＜ファイル名|ディレクトリ名＞

# リポジトリからだけ削除したいとき（ローカル環境にはファイルが残る）
git rm --cashed

#削除について補足
削除コマンド実行時は、削除コマンドが実行されたことがステージに記録される。ステージの内容を反映する場合は、commitを行う

#とりあえずの戻し方
git reset HEAD index.html
git checkout index.html # リポジトリからのチェックアウト

# ファイルの移動を記録する
git mv <旧ファイル> <新ファイル>

'''
